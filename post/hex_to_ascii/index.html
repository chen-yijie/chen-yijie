char *hex_to_ascii(char *dest, const char src[], int size) { char *ret = dest; int i;

for (i = 0; i < size; i++)
{
	\*dest++ = "0123456789ABCDEF"\[(unsigned char)src\[i\] >> 4\];
	\*dest++ = "0123456789ABCDEF"\[src\[i\] & 0xf\];
	\*dest++ = ' ';
}
\*dest = 0;

return ret;

}

return ret;
